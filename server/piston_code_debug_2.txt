def findWords(words):
    row1 = set("qwertyuiop")
    row2 = set("asdfghjkl")
    row3 = set("zxcvbnm")
    
    result = []
    for word in words:
        letters = set(word)  # BUG: missing .lower() here
        if letters <= row1 or letters <= row2 or letters <= row3:
            result.append(word)
    
    return result
